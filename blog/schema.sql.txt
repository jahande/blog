CREATE TABLE article (id INT AUTO_INCREMENT, like_ INT UNSIGNED DEFAULT 0 NOT NULL, dislike INT UNSIGNED DEFAULT 0 NOT NULL, title VARCHAR(255) NOT NULL, content TEXT NOT NULL, created_at TIMESTAMP ON UPDATE CURRENT_TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP, PRIMARY KEY(id)) ENGINE = INNODB;
CREATE TABLE comment (id INT AUTO_INCREMENT, like_ INT UNSIGNED DEFAULT 0 NOT NULL, dislike INT UNSIGNED DEFAULT 0 NOT NULL, article_id INT, author VARCHAR(255) NOT NULL, created_at TIMESTAMP ON UPDATE CURRENT_TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP, content TEXT NOT NULL, INDEX article_id_idx (article_id), PRIMARY KEY(id)) ENGINE = INNODB;
CREATE TABLE tag (id INT AUTO_INCREMENT, article_id INT, name VARCHAR(255), PRIMARY KEY(id, article_id)) ENGINE = INNODB;
CREATE TABLE user (id INT AUTO_INCREMENT, user_name VARCHAR(255), password VARCHAR(255), credential VARCHAR(255), PRIMARY KEY(id, user_name)) ENGINE = INNODB;
ALTER TABLE comment ADD CONSTRAINT comment_article_id_article_id FOREIGN KEY (article_id) REFERENCES article(id) ON DELETE CASCADE;
ALTER TABLE tag ADD CONSTRAINT tag_article_id_article_id FOREIGN KEY (article_id) REFERENCES article(id) ON DELETE CASCADE;
